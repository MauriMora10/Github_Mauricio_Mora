…o crear un nuevo repositorio en la línea de comandos
echo "# Github_Mauricio_Mora" >> README.md 
git init 
git add README.md 
git commit -m "primer commit" 
git branch -M main 
git remote add origin https://github.com/MauriMora10/Github_Mauricio_Mora.git
 git push -u origin main
 
…o enviar un repositorio existente desde la línea de comandos
git remoto agregar origen https://github.com/MauriMora10/Github_Mauricio_Mora.git
 git rama -M principal 
git push -u origen principal
